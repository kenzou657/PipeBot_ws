## 2. 代码详细解读

### 2.1 初始化流程

```python
class BluetoothReceiverNode(Node):
    def __init__(self):
        super().__init__('bluetooth_receiver_node')
        
        # ========== 第一步：声明参数 ==========
        # 这些参数可以通过 ROS2 配置文件修改，无需改代码
        self.declare_parameter('bt_name', 'PipeBot Control')
        self.declare_parameter('service_uuid', '12345678-1234-5678-1234-56789ABCDEF0')
        self.declare_parameter('rx_char_uuid', '12345678-1234-5678-1234-56789ABCDEF1')
        self.declare_parameter('tx_char_uuid', '12345678-1234-5678-1234-56789ABCDEF2')
        
        # 获取参数值
        self.service_uuid = self.get_parameter('service_uuid').value
        
        # ========== 第二步：创建 ROS2 通信接口 ==========
        # 发布者：发送接收到的命令
        self.command_pub = self.create_publisher(
            UInt8MultiArray,        # 消息类型：无符号 8 位整数数组
            'bluetooth/command',    # 话题名称
            10                      # 队列大小
        )
        
        # 订阅者：接收其他节点的反馈数据
        self.feedback_sub = self.create_subscription(
            UInt8MultiArray,
            'bluetooth/feedback',
            self.feedback_callback,  # 回调函数
            10
        )
        
        # ========== 第三步：启动蓝牙线程 ==========
        # 为什么要用线程？因为 BLE 主循环是阻塞的
        self.bt_thread = threading.Thread(
            target=self._run_ble_server,
            daemon=True  # 守护线程：主程序退出时自动关闭
        )
        self.bt_thread.start()
```

### 2.2 BLE 外设设置

```python
def _setup_ble_peripheral(self):
    """设置 BLE 外设的详细步骤"""
    
    # 步骤 1：创建外设对象
    # adapter_address：蓝牙适配器地址（树莓派的蓝牙芯片地址）
    # local_name：设备名称（手机扫描时看到的名字）
    self.peripheral = peripheral.Peripheral(
        adapter_address='88:A2:9E:4B:BC:E6',
        local_name='PipeBot Control'
    )
    
    # 步骤 2：添加服务
    # srv_id：服务 ID（本地标识，不是 UUID）
    # uuid：服务的全局唯一标识符
    # primary：是否为主服务（通常为 True）
    self.peripheral.add_service(
        srv_id=1,
        uuid='12345678-1234-5678-1234-56789ABCDEF0',
        primary=True
    )
    
    # 步骤 3：添加接收特征值（手机 → 树莓派）
    # chr_id：特征值 ID（本地标识）
    # flags：特征值属性
    #   - 'write'：支持写入
    #   - 'write-without-response'：写入无需等待响应
    # write_callback：当手机写入时调用的函数
    self.peripheral.add_characteristic(
        srv_id=1,
        chr_id=2,
        uuid='12345678-1234-5678-1234-56789ABCDEF1',
        value=b'',
        notifying=False,
        flags=['write', 'write-without-response'],
        write_callback=self._on_write_request
    )
    
    # 步骤 4：添加发送特征值（树莓派 → 手机）
    # flags：
    #   - 'read'：支持读取
    #   - 'notify'：支持通知（主动推送）
    self.peripheral.add_characteristic(
        srv_id=1,
        chr_id=3,
        uuid='12345678-1234-5678-1234-56789ABCDEF2',
        value=b'',
        notifying=False,
        flags=['read', 'notify']
    )
    
    # 步骤 5：保存发送特征值的引用
    # 后续需要通过这个引用发送数据给手机
    for char in self.peripheral.characteristics:
        if '0003' in char.get_path():  # chr_id=3
            self.tx_characteristic = char
            break
```

### 2.3 接收命令流程

```python
def _on_write_request(self, value, options):
    """
    当手机写入数据时触发
    
    参数说明：
    - value：手机发送的数据（整数列表）
    - options：写入选项（通常不用）
    """
    try:
        # 将数据转换为列表
        data = list(value)
        
        # 创建 ROS2 消息
        command_msg = UInt8MultiArray()
        command_msg.data = data
        
        # 发布到 /bluetooth/command 话题
        # 其他节点可以订阅这个话题来接收命令
        self.command_pub.publish(command_msg)
        
        # 记录日志（便于调试）
        hex_str = ' '.join([f'{b:02X}' for b in data])
        self.get_logger().info(f'接收到命令: {hex_str}')
    
    except Exception as e:
        self.get_logger().error(f'处理命令失败: {e}')
```

### 2.4 发送反馈流程

```python
def feedback_callback(self, msg):
    """
    当其他节点发布到 /bluetooth/feedback 时触发
    
    流程：
    1. 接收 ROS2 消息
    2. 更新 BLE 特征值
    3. 发送通知给手机
    """
    if len(msg.data) > 0 and self.tx_characteristic is not None:
        try:
            # 步骤 1：转换数据格式
            data_list = list(msg.data)
            
            # 步骤 2：更新特征值
            # set_value()：设置特征值的新值
            self.tx_characteristic.set_value(data_list)
            
            # 步骤 3：发送通知
            # PropertiesChanged()：D-Bus 信号，通知客户端属性已改变
            # 这会触发订阅该特征值的手机接收通知
            self.tx_characteristic.PropertiesChanged(
                'org.bluez.GattCharacteristic1',  # 接口名称
                {'Value': data_list},              # 改变的属性
                []                                 # 无效的属性
            )
            
            # 步骤 4：记录日志
            hex_str = ' '.join([f'{b:02X}' for b in data_list[:20]])
            self.get_logger().info(f'已发送反馈数据: {len(data_list)} 字节 [{hex_str}]')
        
        except Exception as e:
            self.get_logger().error(f'发送反馈失败: {e}')
```

## 3. 数据格式说明

### 3.1 UInt8MultiArray 消息格式

```python
from std_msgs.msg import UInt8MultiArray

# 创建消息
msg = UInt8MultiArray()

# 设置数据（无符号 8 位整数数组）
msg.data = [0x01, 0x02, 0x03, 0x04, 0x05]

# 或者
msg.data = [1, 2, 3, 4, 5]

# 发布
publisher.publish(msg)
```

### 3.2 十六进制与十进制转换

```python
# 十六进制 → 十进制
0x01 = 1
0xFF = 255
0xAA = 170

# 十进制 → 十六进制
1 = 0x01
255 = 0xFF
170 = 0xAA

# Python 中的转换
hex(255)      # '0xff'
int('0xFF', 16)  # 255
```

## 6. 常见错误及解决方案

| 错误 | 原因 | 解决方案 |
|------|------|--------|
| `AttributeError: 'Characteristic' object has no attribute 'notify'` | 使用了错误的方法名 | 使用 `PropertiesChanged()` 代替 |
| `ModuleNotFoundError: No module named 'bluezero'` | 未安装 bluezero 库 | `pip install bluezero` |
| `PermissionError: Cannot access /dev/mem` | 权限不足 | 使用 `sudo` 运行 |
| 手机无法发现设备 | 蓝牙未启用或广播未启动 | 检查 `_start_advertising()` 是否被调用 |
| 接收不到命令 | 回调函数未被触发 | 检查 GLib 主循环是否运行 |

## 7. 性能优化建议

### 7.1 减少日志输出

```python
# 改用 debug 级别（默认不显示）
self.get_logger().debug(f'详细信息: {data}')

# 只在需要时输出
if len(data) > 100:
    self.get_logger().info(f'大数据包: {len(data)} 字节')
```

### 7.2 优化数据处理

```python
# 不好的做法：每次都转换
for byte in msg.data:
    hex_str += f'{byte:02X} '

# 好的做法：只在需要时转换
if self.get_logger().isEnabledFor(logging.INFO):
    hex_str = ' '.join([f'{b:02X}' for b in msg.data[:20]])
    self.get_logger().info(f'数据: {hex_str}')
```
